Python Turtle Script

This page shows the Python turtle code for "A starry night on a farm".

from turtle import *
#Enzo Domenico October 5th "A starry night on a farm"
#change background to blue
bgcolor("#3A4B65")
bgcolor()
(128.0, 0.0, 128.0)
#get pen ready to draw the house
width (8)
color('black')
fillcolor ("#4A310D")

speed(2)
#create the square part of the house
begin_fill()
for i in range(4):
    forward(100)
    left(90)
end_fill()
#position pen to draw roof
up()
goto (100,100)
down()
color('black')
fillcolor ('red')

speed(6)

begin_fill()
#draw triange roof
for i in range (3):
 left (120)
 forward (100)
end_fill()

speed(10)
#get pen positioned to draw the grass
up()
goto (-400,0)
down() 
#draw grass
fillcolor('dark green')
begin_fill()
for i in range(4):
    forward(800)
    right(90)
end_fill()
